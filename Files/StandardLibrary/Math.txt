func intDiv(nr a, nr b) -> nr:
	return floor(a / b)

func abs(nr a) -> nr {
	if a < 0:
		return a * -1
	return a
}

func floor(nr a) -> nr {
	if a > 0 and rnd(a) > a:
		return rnd(a - 1)
	elif a < 0 and rnd(a) < a:
		return rnd(a + 1)
	return rnd(a)
}

func ceil(nr a) -> nr {
	if a > 0 and rnd(a) < a:
		return rnd(a + 1)
	elif a < 0 and rnd(a) > a:
		return rnd(a - 1)
	return rnd(a)
}
